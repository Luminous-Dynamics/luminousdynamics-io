Finding Renewable Energy Sites

Search for solar and wind energy sites within a specific geographic area, with filtering by capacity and project status.

Terra Atlas REST API Beginner 
import { LuminousClient } from '@luminous/api-client';

const client = new LuminousClient({
  apiKey: process.env.LUMINOUS_API_KEY
});

async function findNearbySolarSites() {
  try {
    const sites = await client.terra.sites.search({
      type: 'solar',
      minCapacity: 1000, // kW
      location: {
        lat: 40.7128,
        lng: -74.0060,
        radius: 50 // km
      },
      status: 'operational',
      limit: 20
    });

    console.log(`Found ${sites.total} solar sites`);

    sites.data.forEach(site => {
      console.log(`${site.name}: ${site.capacity}kW`);
      console.log(`  Location: ${site.location.city}, ${site.location.state}`);
      console.log(`  ROI: ${site.metrics.roi}`);
    });

    return sites;
  } catch (error) {
    console.error('Error searching sites:', error.message);
    throw error;
  }
}

package main

import (
    "fmt"
    "os"
    "github.com/luminous-dynamics/go-client"
)

func main() {
    client := luminous.NewClient(os.Getenv("LUMINOUS_API_KEY"))

    sites, err := client.Terra.Sites.Search(&luminous.SiteSearchParams{
        Type:        "solar",
        MinCapacity: 1000, // kW
        Location: &luminous.Location{
            Lat:    40.7128,
            Lng:    -74.0060,
            Radius: 50, // km
        },
        Status: "operational",
        Limit:  20,
    })

    if err != nil {
        fmt.Printf("Error searching sites: %v\n", err)
        return
    }

    fmt.Printf("Found %d solar sites\n", sites.Total)

    for _, site := range sites.Data {
        fmt.Printf("%s: %dkW\n", site.Name, site.Capacity)
        fmt.Printf("  Location: %s, %s\n", site.Location.City, site.Location.State)
        fmt.Printf("  ROI: %s\n", site.Metrics.ROI)
    }
}

Robust Error Handling

Implement proper error handling with retry logic and exponential backoff for production applications.

Terra Atlas Best Practices Advanced 
import { LuminousClient, LuminousError } from '@luminous/api-client';

const client = new LuminousClient({
  apiKey: process.env.LUMINOUS_API_KEY,
  retryConfig: {
    maxRetries: 3,
    retryDelay: 1000
  }
});

async function retryWithBackoff(fn, maxRetries = 3) {
  for (let attempt = 0; attempt <= maxRetries; attempt++) {
    try {
      return await fn();
    } catch (error) {
      if (attempt === maxRetries) throw error;

      // Don't retry client errors (4xx)
      if (error instanceof LuminousError && error.statusCode < 500) {
        throw error;
      }

      const delay = Math.min(1000 * Math.pow(2, attempt), 10000);
      const jitter = Math.random() * 1000;

      console.log(`Retry ${attempt + 1}/${maxRetries} after ${delay + jitter}ms`);
      await new Promise(resolve => setTimeout(resolve, delay + jitter));
    }
  }
}

async function fetchSiteWithRetry(siteId) {
  try {
    const site = await retryWithBackoff(async () => {
      return await client.terra.sites.get(siteId);
    });

    return site;
  } catch (error) {
    if (error instanceof LuminousError) {
      switch (error.statusCode) {
        case 401:
          console.error('Authentication failed - check your API key');
          break;
        case 404:
          console.error(`Site ${siteId} not found`);
          break;
        case 429:
          console.error('Rate limit exceeded - slow down requests');
          break;
        default:
          console.error(`API error: ${error.message}`);
      }
    } else {
      console.error('Unexpected error:', error);
    }
    throw error;
  }
}

export { fetchSiteWithRetry };
from luminous import LuminousClient, LuminousError
import time
import random
import os

client = LuminousClient(api_key=os.getenv('LUMINOUS_API_KEY'))

def retry_with_backoff(func, max_retries=3):
    for attempt in range(max_retries + 1):
        try:
            return func()
        except Exception as error:
            if attempt == max_retries:
                raise

            # Don't retry client errors (4xx)
            if isinstance(error, LuminousError) and error.status_code < 500:
                raise

            delay = min(1000 * (2 ** attempt), 10000) / 1000
            jitter = random.random()

            print(f"Retry {attempt + 1}/{max_retries} after {delay + jitter}s")
            time.sleep(delay + jitter)

def fetch_site_with_retry(site_id):
    try:
        site = retry_with_backoff(
            lambda: client.terra.sites.get(site_id)
        )
        return site
    except LuminousError as error:
        if error.status_code == 401:
            print('Authentication failed - check your API key')
        elif error.status_code == 404:
            print(f'Site {site_id} not found')
        elif error.status_code == 429:
            print('Rate limit exceeded - slow down requests')
        else:
            print(f'API error: {error.message}')
        raise
    except Exception as error:
        print(f'Unexpected error: {error}')
        raise

💡 Best Practices

  • Always implement exponential backoff for retries
  • Add jitter to prevent thundering herd problem
  • Don't retry 4xx errors (client errors) - only 5xx (server errors)
  • Respect rate limits (429 status code)
  • Log errors appropriately for debugging

Natural Language NixOS Operations

Use natural language to perform system operations on NixOS with semantic understanding and safety checks.

Luminous Nix CLI Intermediate 
from luminous_nix import NixAssistant, SafetyLevel

# Initialize with safety checks enabled
nix = NixAssistant(
    safety_level=SafetyLevel.INTERACTIVE,
    dry_run=False
)

# Install packages with natural language
result = nix.ask("install firefox and thunderbird")

if result.success:
    print(f"✓ {result.message}")
    print(f"Packages installed: {', '.join(result.packages)}")
else:
    print(f"✗ Error: {result.error}")

# Query system configuration
info = nix.ask("show me which version of python is installed")
print(info.response)

# Rebuild configuration
rebuild = nix.ask("rebuild my system configuration")
if rebuild.requires_confirmation:
    print(f"This will: {rebuild.description}")
    if input("Continue? (y/n): ").lower() == 'y':
        rebuild.execute()

luminous-nix ask --dry-run "update all my packages"

# Configuration management

💡 Safety Features

  • All destructive operations require confirmation by default
  • Dry-run mode lets you preview changes before applying
  • Semantic understanding prevents ambiguous or unsafe commands
  • Automatic rollback on failure

Real-time Updates with WebSockets

Subscribe to real-time updates using WebSockets for live data streaming and event notifications.

Sacred Bridge WebSocket Advanced 
import { SacredBridgeClient } from '@luminous/sacred-bridge';

const bridge = new SacredBridgeClient({
  apiKey: process.env.LUMINOUS_API_KEY,
  autoReconnect: true,
  heartbeatInterval: 30000
});

// Connect to WebSocket
await bridge.connect();

// Subscribe to site updates
bridge.subscribe('terra.sites.updates', (event) => {
  console.log('Site updated:', event.data);

  // Handle different event types
  switch (event.type) {
    case 'site.created':
      console.log(`New site: ${event.data.name}`);
      break;
    case 'site.updated':
      console.log(`Updated: ${event.data.name}`);
      break;
    case 'site.status_changed':
      console.log(`Status: ${event.data.status}`);
      break;
  }
});

// Subscribe to system events
bridge.subscribe('system.events', (event) => {
  console.log('System event:', event.type);
});

// Handle connection events
bridge.on('connected', () => {
  console.log('✓ Connected to Sacred Bridge');
});

bridge.on('disconnected', () => {
  console.log('✗ Disconnected');
});

bridge.on('error', (error) => {
  console.error('WebSocket error:', error);
});

// Graceful shutdown
process.on('SIGINT', async () => {
  await bridge.disconnect();
  process.exit(0);
});
from luminous.sacred_bridge import SacredBridgeClient
import asyncio
import os

bridge = SacredBridgeClient(
    api_key=os.getenv('LUMINOUS_API_KEY'),
    auto_reconnect=True,
    heartbeat_interval=30
)

async def handle_site_update(event):
    print(f'Site updated: {event["data"]}')

    event_type = event['type']
    if event_type == 'site.created':
        print(f'New site: {event["data"]["name"]}')
    elif event_type == 'site.updated':
        print(f'Updated: {event["data"]["name"]}')
    elif event_type == 'site.status_changed':
        print(f'Status: {event["data"]["status"]}')

async def handle_system_event(event):
    print(f'System event: {event["type"]}')

async def main():
    # Connect to WebSocket
    await bridge.connect()

    # Subscribe to updates
    await bridge.subscribe('terra.sites.updates', handle_site_update)
    await bridge.subscribe('system.events', handle_system_event)

    # Handle connection events
    bridge.on('connected', lambda: print('✓ Connected to Sacred Bridge'))
    bridge.on('disconnected', lambda: print('✗ Disconnected'))
    bridge.on('error', lambda e: print(f'WebSocket error: {e}'))

    # Keep running
    try:
        await bridge.wait_forever()
    except KeyboardInterrupt:
        await bridge.disconnect()

if __name__ == '__main__':
    asyncio.run(main())

💡 WebSocket Best Practices

  • Enable auto-reconnect for production applications
  • Implement heartbeat to detect connection issues
  • Handle connection, disconnection, and error events
  • Use message queues for high-volume events
  • Implement exponential backoff for reconnections
