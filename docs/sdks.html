    SDKs & Libraries

    Official SDKs and community-maintained libraries for integrating Luminous Dynamics APIs into your applications. Choose from multiple programming languages with idiomatic interfaces.

    📅 Last updated: November 2025 ⏱️ 10 minute read

    Official SDKs

    Luminous Dynamics maintains official SDKs for the most popular programming languages. All official SDKs are open-source, thoroughly tested, and actively maintained.

    JavaScript / Node.js

    Full-featured SDK for Node.js and browser environments with TypeScript support.

    bash 
    # npm
npm install @luminous/api-client

# yarn
yarn add @luminous/api-client

# pnpm
pnpm add @luminous/api-client

    Quick Start

    import { LuminousClient } from '@luminous/api-client';

const client = new LuminousClient({
  apiKey: process.env.LUMINOUS_API_KEY
});

// Terra Atlas - Find solar sites
const sites = await client.terra.sites.list({
  type: 'solar',
  minCapacity: 1000
});

// Luminous Nix - Analyze consciousness
const analysis = await client.nix.analyze({
  data: [0.45, 0.78, 0.92],
  depth: 7
});

// Sacred Bridge - Integrate patterns
const integration = await client.bridge.integrate({
  patterns: [...patterns],
  mode: 'balanced'
});

    Features

    • ✅ TypeScript definitions included
    • ✅ Promise-based async/await API
    • ✅ Automatic retry with exponential backoff
    • ✅ Request/response interceptors
    • ✅ Built-in error handling
    • ✅ Rate limit tracking
    • ✅ Webhook signature verification
    ℹ️ Documentation

    Full documentation: github.com/Luminous-Dynamics/js-sdk

    Python

    Pythonic SDK with async support and comprehensive type hints.

    bash 
    # pip
pip install luminous-api

# poetry
poetry add luminous-api

    Quick Start

    from luminous import LuminousClient

client = LuminousClient(api_key=os.getenv('LUMINOUS_API_KEY'))

# Terra Atlas
sites = client.terra.sites.list(
    type='solar',
    min_capacity=1000
)

# Luminous Nix
analysis = client.nix.analyze(
    data=[0.45, 0.78, 0.92],
    depth=7
)

# Async support
import asyncio

async def main():
    async with LuminousClient(api_key=api_key) as client:
        sites = await client.terra.sites.list(type='solar')
        print(sites)

asyncio.run(main())

    Features

    • ✅ Full type hints (mypy compatible)
    • ✅ Sync and async APIs
    • ✅ Pydantic models for validation
    • ✅ Automatic pagination
    • ✅ Context manager support
    • ✅ Logging integration

    Go

    Idiomatic Go SDK with comprehensive error handling. 

    go get github.com/luminous-dynamics/go-sdk

    Quick Start

    package main

import (
    "context"
    "log"
    luminous "github.com/luminous-dynamics/go-sdk"
)

func main() {
    client := luminous.NewClient(os.Getenv("LUMINOUS_API_KEY"))

    // Terra Atlas
    sites, err := client.Terra.Sites.List(context.Background(), &luminous.SiteListParams{
        Type: luminous.String("solar"),
        MinCapacity: luminous.Int64(1000),
    })
    if err != nil {
        log.Fatal(err)
    }

    for _, site := range sites.Data {
        log.Printf("Site: %s, Capacity: %d kW", site.Name, site.Capacity)
    }
}

    Features

    • ✅ Context-aware operations
    • ✅ Comprehensive error types
    • ✅ Automatic retries
    • ✅ Structured logging
    • ✅ Mock client for testing

    Rust

    High-performance Rust SDK with strong type safety. 

    # Cargo.toml
[dependencies]
luminous-api = "0.1"
tokio = { version = "1", features = ["full"] }

    Quick Start

    use luminous_api::{Client, SiteType};

#[tokio::main]
async fn main() -> Result<(), Box> {
    let client = Client::new(std::env::var("LUMINOUS_API_KEY")?);

    let sites = client.terra().sites()
        .list()
        .site_type(SiteType::Solar)
        .min_capacity(1000)
        .send()
        .await?;

    for site in sites.data {
        println!("Site: {}, Capacity: {} kW", site.name, site.capacity);
    }

    Ok(())
}

    Features

    • ✅ Builder pattern for requests
    • ✅ Strong type safety
    • ✅ Async/await with tokio
    • ✅ Zero-copy deserialization
    • ✅ Comprehensive error handling

    Community SDKs

    Community-maintained SDKs for additional languages. While not officially supported, these libraries are widely used and actively maintained.

    Ruby

    # Gemfile
gem 'luminous-api'

# Or install directly
gem install luminous-api
    require 'luminous'

client = Luminous::Client.new(api_key: ENV['LUMINOUS_API_KEY'])

sites = client.terra.sites.list(type: 'solar', min_capacity: 1000)
sites.each do |site|
  puts "#{site.name}: #{site.capacity} kW"
end

    PHP

    composer require luminous/api-client

    Java

    <!-- Maven -->
<dependency>
    <groupId>io.luminousdynamics</groupId>
    <artifactId>luminous-api</artifactId>
    <version>1.0.0</version>
</dependency>
    import io.luminousdynamics.LuminousClient;
import io.luminousdynamics.models.Site;

public class Example {
    public static void main(String[] args) {
        LuminousClient client = new LuminousClient(
            System.getenv("LUMINOUS_API_KEY")
        );

        List<Site> sites = client.terra().sites()
            .list()
            .type("solar")
            .minCapacity(1000)
            .execute();

        sites.forEach(site ->
            System.out.printf("%s: %d kW%n", site.getName(), site.getCapacity())
        );
    }
}
